pipeline {
    agent any

    environment {
        IMAGE_NAME = "hardened-python-hello"
        CONTAINER_NAME = "secure-app-pipeline"
    }

    stages {
        stage('1. Code Checkout') {
            steps {
                echo 'Pulling the latest code from GitHub...'
                checkout scm
            }
        }

        stage('2. Lint & Static Analysis') {
            steps {
                echo 'Checking Dockerfile for basic security issues...'
                // Optional: Runs Hadolint if installed on the Jenkins server to find Dockerfile flaws
                // sh 'hadolint Dockerfile'
            }
        }

        stage('3. Build Hardened Image') {
            steps {
                echo "Compiling multi-stage Docker image: ${IMAGE_NAME}..."
                sh "docker build -t ${IMAGE_NAME} ."
            }
        }

        stage('4. Security & Health Audits') {
            steps {
                echo 'Spinning up the container sandbox for validation tests...'
                // Run the container in detached mode
                sh "docker run -d -p 8000:8000 --name ${CONTAINER_NAME} ${IMAGE_NAME}"
                
                // Give the Healthcheck 10 seconds to initial boot before checking status
                sleep time: 10, unit: 'SECONDS'

                script {
                    try {
                        // Audit 1: Verify the running user is unprivileged
                        echo 'Auditing execution boundary...'
                        sh "docker exec ${CONTAINER_NAME} whoami"

                        // Audit 2: Verify the orchestration health probe status
                        echo 'Auditing telemetry health...'
                        sh "docker ps --filter name=${CONTAINER_NAME} | grep -q '(healthy)'"
                        
                        echo '✅ All infrastructure validations passed successfully!'
                    } finally {
                        // This block ALWAYS runs, ensuring we don't leave zombie containers running on the server
                        echo 'Cleaning up the testing sandbox...'
                        sh "docker stop ${CONTAINER_NAME} || true"
                        sh "docker rm ${CONTAINER_NAME} || true"
                    }
                }
            }
        }
    }

    post {
        success {
            echo '🎉 Pipeline successfully completed! The container is secure and production-ready.'
        }
        failure {
            echo '❌ Pipeline failed! Review the logs above to check which security validation audit failed.'
        }
    }
}
